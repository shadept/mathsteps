Get started

Up and running in seconds

example.js
const mathsteps = require('mathsteps');

// Simplify an expression step by step
const steps = mathsteps.simplifyExpression('2x + 2x + x + x');

steps.forEach(step => {
  console.log(step.changeType);  // e.g. ADD_POLYNOMIAL_TERMS
  console.log(step.newNode.toString()); // e.g. 6x
});

// Solve an equation
const solution = mathsteps.solveEquation('2x + 3x = 35');
